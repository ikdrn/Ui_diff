5. Toast Notifications 
function showToast(type, title, message) {
  const container = document.getElementById('toast-container');
  const toast = document.createElement('div');
  toast.className = `toast toast-${type}`;
  toast.innerHTML = `
    <div class="toast-icon">...svg...</div>
    <div class="toast-body">
      <div class="toast-title">${title}</div>
      <div class="toast-msg">${message}</div>
    </div>
    <button class="toast-close">×</button>
  `;
  container.appendChild(toast);
  setTimeout(() => removeToast(toast), 4000);
}

9. Alert with Progress Bar (auto-dismiss)
Upload complete! All 5 files were uploaded successfully.
Syncing... Your data is being synchronized in the background. 
/* The progress bar shrinks from 100% → 0% over N seconds */
function startProgressAlert(alertEl, barEl, duration = 5000) {
  barEl.style.transition = `width ${duration}ms linear`;
  barEl.style.width = '100%';
  requestAnimationFrame(() => {
    requestAnimationFrame(() => { barEl.style.width = '0%'; });
  });
  setTimeout(() => {
    alertEl.classList.add('hiding');
    setTimeout(() => alertEl.remove(), 300);
  }, duration);
}
