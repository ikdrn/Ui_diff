5. Animated Particles Hero

Float Through Space

CSS-animated floating particles create depth and life. Each dot has a randomised size, position, colour and animation duration.

Get Floating → 
@keyframes float-particle {
  0%, 100% { transform: translateY(0); opacity: .7; }
  50%       { transform: translateY(-20px); opacity: 1; }
}
.particle {
  position: absolute;
  border-radius: 50%;
  animation: float-particle var(--dur) ease-in-out var(--delay) infinite;
}

/* JS: generate particles */
const bg = document.getElementById('particles-bg');
for (let i = 0; i < 40; i++) {
  const s = document.createElement('span');
  s.className = 'particle';
  const size = Math.random() * 12 + 4;
  const colors = ['#6366f1','#8b5cf6','#06b6d4','#10b981','#f59e0b'];
  s.style.cssText = `
    width:${size}px; height:${size}px;
    left:${Math.random()*100}%; top:${Math.random()*100}%;
    background:${colors[Math.floor(Math.random()*colors.length)]};
    opacity:${Math.random()*.5+.2};
    --dur:${(Math.random()*4+2).toFixed(1)}s;
    --delay:-${(Math.random()*5).toFixed(1)}s;
  `;
  bg.appendChild(s);
}

7. Typed Text Hero

Hello, I am a

Creative

A pure Vanilla JS typing animation cycles through a list of words — no library needed. Configurable speed, words and loop behaviour.

Hire Me → View Portfolio 
const target = document.getElementById('typed-target');
const words = ['Developer', 'Designer', 'Creator'];
let i = 0, j = 0, isDeleting = false;

function type() {
  const word = words[i % words.length];
  target.textContent = isDeleting
    ? word.substring(0, j--)
    : word.substring(0, j++);

  if (!isDeleting && j > word.length) {
    isDeleting = true;
    setTimeout(type, 1500);
    return;
  }
  if (isDeleting && j < 0) {
    isDeleting = false;
    i++;
  }
  setTimeout(type, isDeleting ? 80 : 120);
}

8. Parallax Scroll Hero
⭐ Scroll to Experience

Depth Through Motion

The background layer translates at half the speed of the content on scroll, creating a classic parallax depth effect — pure JS, no libraries.

Explore →
↓ Scroll down 
const heroBg = document.getElementById('parallax-bg');
const heroWrap = document.getElementById('parallax-wrap');

window.addEventListener('scroll', () => {
  const rect = heroWrap.getBoundingClientRect();
  if (rect.bottom < 0 || rect.top > window.innerHeight) return;
  const offset = -rect.top * 0.4; // 40% parallax speed
  heroBg.style.transform = `translateY(${offset}px)`;
}, { passive: true });

11. 3D Perspective Hero
◕ Perspective Transform

Design in Three
Dimensions

CSS perspective and rotateX/rotateY transforms give the card a tilt. Hover over it to watch the card flatten back to face — or move your mouse for live 3D tracking.

Explore → Learn More 
<style>
.hero-3d {
  perspective: 1000px;
}
.hero-3d__stage {
  transform: rotateX(6deg) rotateY(-4deg);
  transform-style: preserve-3d;
  transition: transform 0.6s ease;
}
.hero-3d__stage:hover {
  transform: rotateX(0deg) rotateY(0deg);
}
.hero-3d__card {
  transform: translateZ(30px);
  background: rgba(255,255,255,0.05);
  backdrop-filter: blur(12px);
  border: 1px solid rgba(255,255,255,0.12);
  border-radius: 24px;
}
</style>

<!-- JS: live mouse tracking -->
<script>
const stage = document.getElementById('hero-3d-stage');
stage.closest('.hero-3d').addEventListener('mousemove', e => {
  const rect = e.currentTarget.getBoundingClientRect();
  const x = ((e.clientX - rect.left) / rect.width  - 0.5) * 20;
  const y = ((e.clientY - rect.top)  / rect.height - 0.5) * -14;
  stage.style.transform = `rotateX(${y}deg) rotateY(${x}deg)`;
});
stage.closest('.hero-3d').addEventListener('mouseleave', () => {
  stage.style.transform = 'rotateX(6deg) rotateY(-4deg)';
});
</script>
