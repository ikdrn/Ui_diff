5. Searchable Dropdown 
<div class="dropdown-menu">
  <div class="dropdown-search-box">
    <input type="text" class="dropdown-search-input"
      placeholder="Search..." id="my-search">
  </div>
  <div id="results-list">
    <!-- フィルタ結果を JS で動的に表示 -->
  </div>
</div>

<script>
  document.getElementById('my-search').addEventListener('input', function(e) {
    var q = e.target.value.toLowerCase();
    document.querySelectorAll('.dropdown-item').forEach(function(item) {
      item.style.display = item.textContent.toLowerCase().includes(q) ? '' : 'none';
    });
  });
</script>

7. Context Menu Style

Right-click (or long-press on mobile) anywhere in the area below:

Right-click here to open the context menu 
/* コンテキストメニュー: position:fixed でカーソル位置に表示 */
element.addEventListener('contextmenu', function(e) {
  e.preventDefault();
  var menu = document.getElementById('context-menu');
  menu.style.left = e.clientX + 'px';
  menu.style.top  = e.clientY + 'px';
  menu.classList.add('open');
});
/* 外部クリックで閉じる */
document.addEventListener('click', function() {
  document.getElementById('context-menu').classList.remove('open');
});
