Modals

Overlay dialogs, drawers, sheets, lightboxes, toasts, and more — all built with CSS transitions and vanilla JS.

1. Basic Modal 
<button onclick="openModal('modal-basic')">Open Modal</button>

<div class="modal-overlay" id="modal-basic">
  <div class="modal-box">
    <div class="modal-header">
      <span class="modal-title">Modal Title</span>
      <button class="modal-close" onclick="closeModal('modal-basic')">&times;</button>
    </div>
    <div class="modal-body">Modal body content goes here.</div>
    <div class="modal-footer">
      <button onclick="closeModal('modal-basic')">Cancel</button>
      <button>Confirm</button>
    </div>
  </div>
</div>

8. Notification Toast 
// Show a toast
showToast('success', 'Success', 'Item saved successfully.');

// Toast container (injected automatically)
<div class="toast-container" id="toast-container"></div>

// Each toast element
<div class="toast toast-success show">
  <span class="toast-icon">✅</span>
  <div class="toast-content">
    <div class="toast-title">Success</div>
    <div class="toast-msg">Item saved successfully.</div>
  </div>
  <button class="toast-close">&times;</button>
  <div class="toast-progress"></div>
</div>
