Pagination

Six pagination patterns — from classic numbered pages to a load-more content list. Click page numbers to toggle active state.

1. Basic Pagination 
<nav class="pagination pag-basic" aria-label="pagination">
  <button class="page-btn disabled" aria-label="Previous">&lsaquo;</button>
  <button class="page-btn active">1</button>
  <button class="page-btn">2</button>
  <button class="page-btn">3</button>
  <button class="page-btn">4</button>
  <button class="page-btn">5</button>
  <button class="page-btn" aria-label="Next">&rsaquo;</button>
</nav>

<style>
.pagination { display:flex; align-items:center; gap:0.25rem; flex-wrap:wrap; }
.pag-basic .page-btn {
  display:inline-flex; align-items:center; justify-content:center;
  min-width:38px; height:38px; padding:0 0.5rem; font-size:0.875rem; font-weight:500;
  color:var(--text-muted); background:var(--surface); border:1px solid var(--border);
  border-radius:var(--radius-sm); cursor:pointer; transition:all 0.2s; user-select:none;
}
.pag-basic .page-btn:hover { background:var(--surface-2); color:var(--text); border-color:var(--gray-400); }
.pag-basic .page-btn.active { background:var(--primary); color:#fff; border-color:var(--primary); }
.pag-basic .page-btn.disabled { opacity:0.45; cursor:not-allowed; pointer-events:none; }
</style>

<script>
document.querySelectorAll('.pagination').forEach(pag => {
  pag.querySelectorAll('.page-btn').forEach(btn => {
    btn.addEventListener('click', () => {
      pag.querySelectorAll('.page-btn').forEach(b => b.classList.remove('active'));
      btn.classList.add('active');
    });
  });
});
</script>

5. Simple Prev / Next
Page 1 of 8 
<nav class="pagination pag-simple" aria-label="pagination">
  <button class="nav-btn" id="prev-btn">&larr; Previous</button>
  <span class="page-info">Page <strong id="cur">1</strong> of 8</span>
  <button class="nav-btn" id="next-btn">Next &rarr;</button>
</nav>

<style>
.pag-simple { gap:0.75rem; }
.pag-simple .nav-btn {
  display:inline-flex; align-items:center; gap:0.4rem; height:42px; padding:0 1.5rem;
  font-size:0.9rem; font-weight:600; color:var(--text); background:var(--surface);
  border:1px solid var(--border); border-radius:9999px; cursor:pointer; transition:all 0.2s;
}
.pag-simple .nav-btn:hover { background:var(--primary); color:#fff; border-color:var(--primary); }
.pag-simple .nav-btn.disabled { opacity:0.4; cursor:not-allowed; pointer-events:none; }
.pag-simple .page-info { font-size:0.875rem; color:var(--text-muted); }
.pag-simple .page-info strong { color:var(--text); }
</style>

<script>
let page = 1;
const total = 8;
const cur = document.getElementById('cur');
const prev = document.getElementById('prev-btn');
const next = document.getElementById('next-btn');

function updateSimple() {
  cur.textContent = page;
  prev.classList.toggle('disabled', page === 1);
  next.classList.toggle('disabled', page === total);
}

prev.addEventListener('click', () => { if (page > 1) { page--; updateSimple(); } });
next.addEventListener('click', () => { if (page < total) { page++; updateSimple(); } });
updateSimple();
</script>

<script>
const moreItems = [
  { num:'04', title:'Vanilla JS DOM', sub:'JavaScript · 6 min read' },
  { num:'05', title:'CSS Animations Deep Dive', sub:'Animation · 7 min read' },
  { num:'06', title:'Responsive Design Tips', sub:'Layout · 5 min read' },
];
let loaded = 0;

document.getElementById('load-more-btn').addEventListener('click', function () {
  this.classList.add('loading');
  setTimeout(() => {
    const item = moreItems[loaded++];
    if (item) {
      const el = document.createElement('div');
      el.className = 'load-more-item';
      el.innerHTML = `<div class="load-more-item-icon">${item.num}</div>
        <div class="load-more-item-text">
          <div class="load-more-item-title">${item.title}</div>
          <div class="load-more-item-sub">${item.sub}</div>
        </div>`;
      document.getElementById('list').appendChild(el);
    }
    this.classList.remove('loading');
    if (loaded >= moreItems.length) {
      this.disabled = true;
      this.querySelector('.load-more-label').textContent = 'All loaded';
    }
  }, 800);
});
</script>
